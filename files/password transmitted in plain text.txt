// Practical: Password transmitted in plain text

#include <iostream>
#include <string>

using namespace std;

int main()
{
    string username;
    string password;

    cout << "Enter username: ";
    cin >> username;

    cout << "Enter password: ";
    cin >> password;

    if (username == "user@gmail.com" && password == "TEENTC23")
    {
        cout << "Entered username and password is correct" << endl;
        cout << "Welcome to Network Security LAB" << endl;
    }
    else
    {
        cout << "Wrong username or password" << endl;
    }

    return 0;
}
