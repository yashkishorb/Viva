// Practical: Encryption and decryption using Caesar cipher

#include <stdio.h>
#include <string.h>

int main()
{
    char str[100];
    int choice;
    int i;
    int key = 3;

    printf("Enter a string: ");
    scanf("%s", str);

    printf("1. Encryption\n");
    printf("2. Decryption\n");
    printf("Enter your choice: ");
    scanf("%d", &choice);

    if (choice == 1)
    {
        for (i = 0; str[i] != '\0'; i++)
        {
            if (str[i] >= 'A' && str[i] <= 'Z')
            {
                str[i] = ((str[i] - 'A' + key) % 26) + 'A';
            }
            else if (str[i] >= 'a' && str[i] <= 'z')
            {
                str[i] = ((str[i] - 'a' + key) % 26) + 'a';
            }
        }

        printf("Encrypted string: %s", str);
    }
    else if (choice == 2)
    {
        for (i = 0; str[i] != '\0'; i++)
        {
            if (str[i] >= 'A' && str[i] <= 'Z')
            {
                str[i] = ((str[i] - 'A' - key + 26) % 26) + 'A';
            }
            else if (str[i] >= 'a' && str[i] <= 'z')
            {
                str[i] = ((str[i] - 'a' - key + 26) % 26) + 'a';
            }
        }

        printf("Decrypted string: %s", str);
    }
    else
    {
        printf("Invalid choice");
    }

    return 0;
}
