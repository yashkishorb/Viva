// Practical: Insecurity of default passwords

#include <stdio.h>
#include <string.h>

int main()
{
    char username[20];
    char password[20];

    printf("Enter your username: ");
    scanf("%s", username);

    printf("Enter your password: ");
    scanf("%s", password);

    if (strcmp(username, "TEENTC") == 0)
    {
        if (strcmp(password, "TEENTC") == 0)
        {
            printf("Welcome Login Success");
        }
        else
        {
            printf("Wrong password");
        }
    }
    else
    {
        printf("User does not exist");
    }

    return 0;
}
