#include <conio.h>
#include <iostream>
#include <string>

using namespace std;

string takePasswdFromUser(char sp = '*')
{
    string passwd = "";
    char ch_ipt;

    while (true)
    {
        ch_ipt = getch();

        if (ch_ipt == '\r' || ch_ipt == '\n')
        {
            cout << endl;
            return passwd;
        }
        else if (ch_ipt == 8 && passwd.length() != 0)
        {
            passwd.erase(passwd.length() - 1);
            cout << "\b \b";
        }
        else if (ch_ipt == 8)
        {
            continue;
        }
        else
        {
            passwd += ch_ipt;
            cout << sp;
        }
    }
}

int main()
{
    string name;
    string input;

    cout << "Enter the Username : ";
    cin >> name;
    cin.ignore();

    cout << "Enter the Password : ";
    input = takePasswdFromUser();

    cout << "Entered Password: " << input << endl;

    return 0;
}
